cmake_minimum_required(VERSION 2.8) 
 
option(PROJECT_NAME "Project name" OFF) 
project(${PROJECT_NAME}) 
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11 -Wall -pthread -m32") 
 
file(GLOB_RECURSE CPPS "*.cpp") 
 
message(STATUS "CMAKE_CURRENT_SOURCE_DIR: ${CMAKE_CURRENT_SOURCE_DIR}") 
message(STATUS "CMAKE_CURRENT_BINARY_DIR: ${CMAKE_CURRENT_BINARY_DIR}") 
 
list(REMOVE_ITEM CPPS "${CMAKE_CURRENT_BINARY_DIR}/CMakeFiles/3.5.1/CompilerIdCXX/CMakeCXXCompilerId.cpp") 
message(STATUS "CPPS: ${CPPS}") 
 
add_executable(${PROJECT_NAME} ${CPPS}) 
 
 
INCLUDE(FindPkgConfig) 
 
PKG_SEARCH_MODULE(PKG_sdl2 REQUIRED sdl2)
PKG_SEARCH_MODULE(PKG_SDL2_image REQUIRED SDL2_image)
PKG_SEARCH_MODULE(PKG_SDL2_ttf REQUIRED SDL2_ttf)
PKG_SEARCH_MODULE(PKG_SDL2_gfx REQUIRED SDL2_gfx)
PKG_SEARCH_MODULE(PKG_SDL2_mixer REQUIRED SDL2_mixer)
PKG_SEARCH_MODULE(PKG_jsoncpp REQUIRED jsoncpp)
 
INCLUDE_DIRECTORIES( ${PKG_sdl2_INCLUDE_DIRS} ${PKG_SDL2_image_INCLUDE_DIRS} ${PKG_SDL2_ttf_INCLUDE_DIRS} ${PKG_SDL2_gfx_INCLUDE_DIRS} ${PKG_SDL2_mixer_INCLUDE_DIRS} ${PKG_jsoncpp_INCLUDE_DIRS} ) 
TARGET_LINK_LIBRARIES( ${PROJECT_NAME} ${PKG_sdl2_LIBRARIES} ${PKG_SDL2_image_LIBRARIES} ${PKG_SDL2_ttf_LIBRARIES} ${PKG_SDL2_gfx_LIBRARIES} ${PKG_SDL2_mixer_LIBRARIES} ${PKG_jsoncpp_LIBRARIES} )
